# Prompt Related
normal="\[\e[0m\]"
red="\[\e[31m\]"
green="\[\e[32m\]"
blue="\[\e[34m\]"

user_host_part="[$red\u$normal@$blue\h$normal]"
path_part="$normal[$green\W$normal]"
export PS1="$user_host_part $path_part \$ "

# Custom Variables
export MAIL="modnosum@gmail.com"
export EDITOR=emacs
export VISUAL=$EDITOR

# Cleaning Files
CLEAN_DEPTH=4
alias clean='find . -maxdepth $CLEAN_DEPTH -type f \( -name "*~" -o -name "#*#" \) -print -delete 2> /dev/null'

# Listing Files
if ls --color &> /dev/null; then
    alias ls='ls --color=auto'
elif ls -G &> /dev/null; then
    alias ls='ls -G'
fi
alias la='ls -al'
alias l='la'
alias ll='ls -l'
alias lh='ls -al --ignore="[A-Za-z]*"'

# Opening Files
if command -v xdg-open &> /dev/null; then
    alias op='xdg-open'
elif command -v open &> /dev/null; then
    alias op='open'
fi

# Git Related
alias gaa='ga --all'
alias ga='git add'
alias gba='gb --all'
alias gb='git branch'
alias gcm='git commit'
alias gcmm='git commit --message'
alias gcof='git checkout --'
alias gco='git checkout'
alias gd='git diff'
alias gh='git push'
alias glg='git log'
alias gl='git pull'
alias gst='git status'

# Emacs Related
alias em='emacs'

# 42 Related
alias 42='bash $HOME/.42FileChecker/42FileChecker.sh'

# Configuration related
alias re_bash='. $HOME/.bashrc'
alias ed_bash='$EDITOR $HOME/.bashrc'
